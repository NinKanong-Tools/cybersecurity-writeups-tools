### What is sublist3r?
Sublister is a tool designed in python and uses OSINT in order to enumerate subdomains of websites. It helps pen-testers in collecting and gathering subdomains for a domain which is their target. In order to fetch the accurate results, sublilster uses many search engines like Google, Yahoo, etc. and even tools like Netcraft, Virustotal, etc.

## Usage

| Short Form | Long Form      | Description |
|------------|----------------|-------------|
| `-d`       | `--domain`     | Domain name to enumerate subdomains of |
| `-b`       | `--bruteforce` | Enable the subbrute bruteforce module |
| `-p`       | `--ports`      | Scan the found subdomains against specific tcp ports |
| `-v`       | `--verbose`    | Enable the verbose mode and display results in realtime |
| `-t`       | `--threads`    | Number of threads to use for subbrute bruteforce |
| `-e`       | `--engines`    | Specify a comma-separated list of search engines |
| `-o`       | `--output`     | Save the results to text file |
| `-h`       | `--help`       | Show the help message and exit |
| `-r`       | `--resolve`    | Resolve IP addresses of discovered subdomains |
| `iL`       | `--fle`        |  Input list of domains from file |

````
#!/usr/bin/env python3
"""
Advanced Subdomain Enumeration Framework
    Usage:
    python3 recon.py targets.txt      # Read from file
    python3 recon.py google.com       # Scan single domain
"""

import os
import subprocess
import sys
from pathlib import Path

GREEN = '\033[92m'
RED = '\033[91m'
NC = '\033[0m' 

def run_cmd(cmd, quiet=False):
    """Execute shell command and log errors"""
    try:
        if not quiet:
            print(f"[*] Running: {cmd}")
        result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
        if result.returncode != 0 and not quiet:
            print(f"{RED}[!] Error: {result.stderr.strip()}{NC}")
        return result.stdout
    except Exception as e:
        print(f"{RED}[!] Failed to run command: {e}{NC}")
        return ""

def main():
    if len(sys.argv) != 2:
        print(f"{RED}[-] Usage: {sys.argv[0]} <targets.txt | domain>{NC}")
        print(f"    Example: {sys.argv[0]} targets.txt")
        print(f"    Example: {sys.argv[0]} google.com")
        sys.exit(1)

    arg = sys.argv[1].strip()

    if Path(arg).is_file():
        print(f"{GREEN}[*] Reading domains from file: {arg}{NC}")
        try:
            with open(arg, 'r') as f:
                domains = [line.strip() for line in f if line.strip() and not line.startswith('#')]
        except Exception as e:
            print(f"{RED}[-] Failed to read file: {e}{NC}")
            sys.exit(1)
    else:
        print(f"{GREEN}[*] Using single domain: {arg}{NC}")
        domains = [arg]

    output_dir = "recon_results"
    Path(output_dir).mkdir(exist_ok=True)

    log_file = Path(output_dir) / "scan.log"
    with open(log_file, "a") as log:
        log.write(f"[*] Recon started at {subprocess.getoutput('date')} for {len(domains)} domains\n")

    for domain in domains:
        print(f"{GREEN}[+] Enumerating subdomains for: {domain}{NC}")
        target_dir = Path(output_dir) / domain
        target_dir.mkdir(exist_ok=True)

        run_cmd(f"sublist3r -d '{domain}' -o '{target_dir}/sublist3r.txt'")
        run_cmd(f"amass enum -d '{domain}' -o '{target_dir}/amass.txt'")
        run_cmd(f"subfinder -d '{domain}' -all -o '{target_dir}/subfinder.txt'")

        print(f"[*] Combining and cleaning subdomains for {domain}...")
        combined_file = target_dir / "all-subs.txt"
        cmd = (
            f"cat '{target_dir}/'*.txt 2>/dev/null "
            f"| grep -E '^[a-zA-Z0-9.-]+\\.{domain}$' "
            f"| sort -u "
            f"> '{combined_file}'"
        )
        run_cmd(cmd)

        if not combined_file.exists() or combined_file.stat().st_size == 0:
            print(f"{RED}[-] No subdomains found for {domain}{NC}")
            continue

        live_file = target_dir / "live-subs.txt"
        print(f"[*] Checking live hosts with httpx...")
        cmd = f"cat '{combined_file}' | httpx -silent -status-code -title -o '{live_file}'"
        run_cmd(cmd)

        if not live_file.exists() or live_file.stat().st_size == 0:
            print(f"{RED}[-] No live hosts found for {domain}{NC}")
        else:
            print(f"{GREEN}[+] Live hosts saved to {live_file}{NC}")

        screenshot_dir = target_dir / "screenshots"
        print(f"[*] Taking screenshots with EyeWitness...")
        cmd = f"cat '{live_file}' | cut -d ' ' -f1 | eyewitness --web --no-confirm -d '{screenshot_dir}'"
        run_cmd(cmd, quiet=True) 
        ports_file = target_dir / "open-ports.txt"
        print(f"[*] Scanning top ports with nmap...")
        cmd = (
            f"nmap -iL '{live_file}' -p 80,443,21,22,8080,8443 -sV -sC -T4 "
            f"-oN '{ports_file}'"
        )
        run_cmd(cmd)

        print(f"{GREEN}[+] Completed: {domain}{NC}")

    print(f"\n{GREEN}[+] All scans complete! Results saved in: {output_dir}{NC}")
    print(f"{GREEN}[*] Check: {output_dir}/{NC}")


if __name__ == "__main__":
    main()
````

### What Hacker or Pentest use sublist3r in real-world:
- Mapping Attack Surface : Hackers use Sublist3r to discover hidden subdomains of a target.
- Finding Entry Points:
   - After enumerating subdomains, hackers probe them for:
        - Open ports (with Nmap)
        - Services (FTP, SSH, RDP, databases)
        - Web apps (login portals, admin dashboards)
- Identifying Forgotten/Unprotected Subdomains:
  - Many companies forget about old test sites like test.company.com or old.company.com.
  - Hackers love these because:
       - They may have default credentials.
       - They may run outdated software with known vulnerabilities.
       - They may bypass security controls (like WAF on the main site).
- Chaining with Other Tools:
  - Sublist3r results are fed into other tools:
       - httprobe → Check which are alive.
       - whatweb / wappalyzer → Detect technologies used.
       - dirsearch / gobuster → Directory brute force.
       - nmap / masscan → Scan for services.
       - metasploit → Exploit weak services.
